  \pgfmathsetmacro{\walllen}{100}

  \coordinate (A) at ($ (O) + (3*\walldeg:\walllen) $);
  \coordinate (B) at ($ (A) + (2*\walldeg:\walllen) $);
  \coordinate (C) at ($ (B) + (1*\walldeg:\walllen) $);
  \coordinate (D) at ($ (C) + (0*\walldeg:\walllen) $);

  \coordinate (mid-A-B) at ($ (A)!0.5!(B) $);
  \coordinate (mid-B-C) at ($ (B)!0.5!(C) $);
  \coordinate (mid-C-D) at ($ (C)!0.5!(D) $);
  \coordinate (mid-D-E) at ($ (D)!0.5!(E) $);
  \coordinate (center) at (mid-C-D |- mid-O-A);

  \draw (mid-O-A) node[left]{1};
  \draw (mid-A-B) node[above left]{2};
  \draw (mid-B-C) node[above left]{3};
  \draw (mid-C-D) node[above]{4};

  \foreach \p in {mid-O-A, mid-A-B, mid-B-C, mid-C-D, mid-D-E} {
    \draw[densely dotted] (\p) -- (center);
  }

  \fill[black] (center) node[below right]{\emph{C}} circle (3);
  \draw[thick] (O) -- (A) -- (B) -- (C) -- (D) -- (E);
